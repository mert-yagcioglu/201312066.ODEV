#1)K-MEANS LE KMELEME
import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import load_breast_cancer
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
from sklearn.decomposition import PCA



data = load_breast_cancer()
X = data.data
y = data.target

#Burda Normalizasyon yapyoruz.
scaler = StandardScaler()
X_normalized = scaler.fit_transform(X)


#K-means
k = 2
kmeans = KMeans(n_clusters=k, random_state=42)

#Model eitiyoruz.
kmeans.fit(X_normalized)

#Kme merkezi ve etiketlerii
cluster_centers = kmeans.cluster_centers_
labels = kmeans.labels_

#PCA ile veriyi iki boyuta drme
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X_normalized)

#Kme merkezlerini iki boyuta drdmz iin iki boyutta gsterme
plt.scatter(X_pca[:, 0], X_pca[:, 1], c=labels, cmap='viridis', alpha=0.5)
plt.scatter(cluster_centers[:, 0], cluster_centers[:, 1], marker='X', s=200, color='red', label='Kme Merkezi')
plt.title('Meme Kanseri Verilerinde K-MEANS Kmeleme')
plt.xlabel('Ana Bileen 1')
plt.ylabel('Ana Bileen 2')
plt.legend()
plt.show()











#2)K-MEDODS LE KMELEME
import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import load_breast_cancer
from sklearn.preprocessing import StandardScaler
from pyclustering.cluster.kmedoids import kmedoids                      #"pip install pyclustering"   ktphanesinin yklenmesi gerek
from pyclustering.utils import calculate_distance_matrix


data = load_breast_cancer()
X = data.data


#Burda Normalizasyon yapyoruz tekrar
scaler = StandardScaler()
X_normalized = scaler.fit_transform(X)

#K-means
k = 2             #Kme says
initial_medoids = np.random.choice(range(len(X_normalized)), k, replace=False)
kmedoids_instance = kmedoids(calculate_distance_matrix(X_normalized), initial_medoids, data_type='distance_matrix')
kmedoids_instance.process()

#Kmeleme sonular
clusters = kmedoids_instance.get_clusters()

#Tm kme merkezlerini belirliyoruz.
medoids = kmedoids_instance.get_medoids()


#PCA ile veriyi iki boyuta dryoruz tekrar
from sklearn.decomposition import PCA
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X_normalized)

#Kme merkezlerini ve kmeleme sonularn grselletiriyoruz.
plt.figure(figsize=(8, 6))
for cluster in clusters:
    plt.scatter(X_pca[cluster, 0], X_pca[cluster, 1], alpha=0.7)

#Bulunan Medoid'leri gsterme
plt.scatter(X_pca[medoids, 0], X_pca[medoids, 1], marker='X', s=200, color='red', label='Medoids')
plt.title('Meme Kanseri Verilerinde K-MEDODS Kmeleme')
plt.xlabel('Ana Bileen 1')
plt.ylabel('Ana Bileen 2')
plt.legend()
plt.show()
